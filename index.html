Assembly: Context Switching (`switch.asm`)

global context_switch
context_switch:
    mov [rdi + 0], rsp    ; Save old stack pointer
    mov rsp, [rsi + 0]    ; Load new stack pointer
    push rbp
    popf                  ; Restore flags register
    ret
